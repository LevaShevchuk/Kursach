Массивы

Разбор типового задания

Условие задачи:

Заполнить одномерный массив B целыми числами из промежутка от -10 до 5.
Найти сумму и среднее арифметическое положительных чётных элементов массива.
Найти произведение отрицательных элементов массива.
Найти минимальный элемент массива.
Отсортировать массив по убыванию.

Программный код:

// блок объявления одномерного массива
Console.WriteLine("Введите размер одномерного массива: ");
int x = Convert.ToInt32(Console.ReadLine());
int[] b = new int[x];
Console.WriteLine("Одномерный массив B:");

Random rand = new Random();
for (int i = 0; i < x; i++)
{
    b[i] = rand.Next(-10, 6);
    Console.Write(b[i] + "\t");
}
Console.WriteLine();

// блок определения суммы и среднего арифметического положительных и чётных элементов одномерного массива
int sum = 0;
int k = 0;
for (int i = 0; i < x; i++)
{
    if (b[i] > 0 && b[i] % 2 == 0)
    {
        sum += b[i];
        k++;
    }
}

if (k > 0) // проверка условия, что в одномерном массиве присутствует хотя бы один положительный чётный элемент
{
    double sr;
    sr = 0;
    sr = sum / k;
    Console.WriteLine("Сумма положительных чётных элементов одномерного массива B равна " + sum);
    Console.WriteLine("Среднее арифметическое положительных чётных элементов одномерного массива B равно " + Math.Round(sr, 2));
}

// вывод сообщения о том, что положительные чётные элементы в массиве отсутствуют
else  
{
    Console.WriteLine("Положительные чётные элементы одномерного массива B отсутствуют!");
}

// блок определения произведения отрицательных элементов одномерного массива
int pr = 1;
for (int i = 0; i < x; i++)
{
    if (b[i] < 0)
    {
        pr *= b[i];
    }
}
Console.WriteLine("Произведение отрицательных элементов одномерного массива B равно " + pr);

// блок определения минимального элемента одномерного массива
int min = b[1];
   for (int i = 0; i < x; i++)
    {
     if (b[i] < min)
   {
      min = b[i];
   }
}
Console.WriteLine("Минимальный элемент одномерного массива B равен " + min);

// блок сортировки массива по убыванию методом «пузырька»
int temp;
for (int i = 0; i < x - 1; i++)
{
    for (int j = i + 1; j < x; j++)
    {
        if (b[i] < b[j])
        {
            temp = b[i];
            b[i] = b[j];
            b[j] = temp;
        }
     }
}

Console.WriteLine("Одномерный массив B, отсортированный в порядке убывания:");
for (int i = 0; i < x; i++)
{
     Console.Write(b[i] + "\t");
}
Console.ReadKey();
